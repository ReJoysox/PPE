const video = document.getElementById('video'); const canvas = document.getElementById('canvas'); const ctx = canvas.getContext('2d'); const status = document.getElementById('status'); const confRange = document.getElementById('conf-range'); const confValLabel = document.getElementById('conf-val'); let session; let isLive = false; const labels = ["Helmet", "No-Helmet", "No-Vest", "Person", "Vest", "Head"]; // 1. ЗАГРУЗКА async function init() { try { session = await ort.InferenceSession.create('./best.onnx', { executionProviders: ['wasm'] }); status.innerText = "✅ МОДЕЛЬ ЗАГРУЖЕНА"; status.style.color = "#22c55e"; } catch (e) { status.innerText = "❌ ОШИБКА ЗАГРУЗКИ"; status.style.color = "#ef4444"; } } // 2. КАМЕРА async function startCamera() { if (!session) return; const stream = await navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } }); video.srcObject = stream; isLive = true; video.onloadedmetadata = () => { canvas.width = video.videoWidth; canvas.height = video.videoHeight; processVideoLoop(); }; } async function processVideoLoop() { if (!isLive) return; await runInference(video); requestAnimationFrame(processVideoLoop); } // 3. ФОТО document.getElementById('file-input').onchange = (e) => { const file = e.target.files[0]; if (!file) return; isLive = false; if (video.srcObject) video.srcObject.getTracks().forEach(t => t.stop()); const img = new Image(); img.src = URL.createObjectURL(file); img.onload = async () => { canvas.width = img.width; canvas.height = img.height; await runInference(img); }; }; // --- ФУНКЦИЯ СИГМОИДЫ (Исправляет баг с 200%) --- function sigmoid(x) { return 1 / (1 + Math.exp(-x)); } // 4. ДВИЖОК async function runInference(source) { if (!session) return; const [w, h] = [640, 640]; const tempCanvas = document.createElement('canvas'); tempCanvas.width = w; tempCanvas.height = h; const tCtx = tempCanvas.getContext('2d'); tCtx.drawImage(source, 0, 0, w, h); const imgData = tCtx.getImageData(0, 0, w, h).data; const input = new Float32Array(3 * w * h); for (let i = 0; i < w * h; i++) { input[i] = imgData[i * 4] / 255.0; input[i + w*h] = imgData[i * 4 + 1] / 255.0; input[i + 2*w*h] = imgData[i * 4 + 2] / 255.0; } const tensor = new ort.Tensor('float32', input, [1, 3, w, h]); const outputs = await session.run({ [session.inputNames[0]]: tensor }); const output = outputs[session.outputNames[0]]; parseAndDraw(output.data, output.dims, source); } // 5. ОТРИСОВКА function parseAndDraw(data, dims, source) { ctx.clearRect(0, 0, canvas.width, canvas.height); ctx.drawImage(source, 0, 0, canvas.width, canvas.height); const threshold = parseInt(confRange.value) / 100; let boxes = []; let numAnchors = dims[2]; let numClasses = dims[1] - 4; // Разбор данных for (let i = 0; i < numAnchors; i++) { let maxProb = 0; let classId = -1; // Ищем лучший класс для этой рамки for (let c = 0; c < numClasses; c++) { // ПРИМЕНЯЕМ СИГМОИДУ ЗДЕСЬ! let raw_val = data[(4 + c) * numAnchors + i]; let prob = sigmoid(raw_val); if (prob > maxProb) { maxProb = prob; classId = c; } } // Если вероятность больше ползунка if (maxProb > threshold) { let cx = data[0 * numAnchors + i]; let cy = data[1 * numAnchors + i]; let bw = data[2 * numAnchors + i]; let bh = data[3 * numAnchors + i]; let x = (cx - bw / 2) * (canvas.width / 640); let y = (cy - bh / 2) * (canvas.height / 640); let w = bw * (canvas.width / 640); let h = bh * (canvas.height / 640); boxes.push({ x, y, w, h, prob: maxProb, classId }); } } // ЖЕСТКИЙ NMS (удаление дубликатов, чтобы не было "накидано") boxes.sort((a, b) => b.prob - a.prob); const finalBoxes = []; while (boxes.length > 0) { const best = boxes.shift(); finalBoxes.push(best); // Удаляем все рамки, которые перекрывают "лучшую" больше чем на 40% boxes = boxes.filter(box => iou(best, box) < 0.4); } // Отрисовка let safeCount = 0; let badCount = 0; finalBoxes.forEach(box => { const labelName = labels[box.classId] || `Class ${box.classId}`; const labelLower = labelName.toLowerCase(); // Если это просто человек - рисуем тонкую белую рамку и идем дальше if (labelLower === "person" || labelLower === "human") { ctx.strokeStyle = "rgba(255,255,255,0.4)"; ctx.lineWidth = 1; ctx.strokeRect(box.x, box.y, box.w, box.h); return; } // Логика цвета const isBad = labelLower.includes("no"); const color = isBad ? "#ef4444" : "#22c55e"; const text = isBad ? "НАРУШИТЕЛЬ" : "В ЗАЩИТЕ"; if (isBad) badCount++; else safeCount++; // Рамка ctx.strokeStyle = color; ctx.lineWidth = 3; ctx.strokeRect(box.x, box.y, box.w, box.h); // Текст с процентами (теперь они от 0 до 100) ctx.fillStyle = color; ctx.fillRect(box.x, box.y > 20 ? box.y - 20 : 0, box.w + 40, 20); ctx.fillStyle = "white"; ctx.font = "bold 13px Arial"; ctx.fillText(`${text} ${Math.round(box.prob * 100)}%`, box.x + 5, box.y > 20 ? box.y - 5 : 15); }); document.getElementById('count-safe').innerText = safeCount; document.getElementById('count-bad').innerText = badCount; } function iou(b1, b2) { const x1 = Math.max(b1.x, b2.x), y1 = Math.max(b1.y, b2.y); const x2 = Math.min(b1.x + b1.w, b2.x + b2.w), y2 = Math.min(b1.y + b1.h, b2.y + b2.h); const inter = Math.max(0, x2 - x1) * Math.max(0, y2 - y1); const union = b1.w * b1.h + b2.w * b2.h - inter; return inter / union; } confRange.addEventListener('input', (e) => { confValLabel.innerText = e.target.value + '%'; }); init();
